CREATE TABLE MASTER_STDKEVI3(

COSTUMER_ID VARCHAR2(25),

COSTUMER_TYPE VARCHAR2(25),
COSTUMER_CATEGORY VARCHAR2(35),

AUTH_STAT CHAR(1) DEFAULT 'U',
RECORD_STAT CHAR(1) DEFAULT 'O',
MOD_NO NUMBER(4,0),
CHECKER_DT_STAMP DATE,
ONCE_AUTH CHAR(1) DEFAULT 'N',
MAKER_ID VARCHAR2(12),
MAKER_DT_STAMP DATE,
CHECKER_ID VARCHAR2(12),

CONSTRAINT PK_MASTER_STDKEVI3 PRIMARY KEY (COSTUMER_ID)
);

select * from MASTER_STDKEVI3;

CREATE SYNONYM MASTERS_STDKEVI3 FOR MASTER_STDKEVI3;

CREATE TABLE DETAIL_STDKEVI3(
    COSTUMER_ID VARCHAR2(25),
    COSTUMER_DETAIL_ID VARCHAR2(25),
    FIELD_DESCRIPTION VARCHAR2(35),
    MANDATORY_VALIDATION VARCHAR2(35),
    PRIMARY KEY (COSTUMER_ID,COSTUMER_DETAIL_ID),
    CONSTRAINT FK_DETAIL_STDKEVI3 FOREIGN KEY (COSTUMER_ID)
    REFERENCES MASTER_STDKEVI3(COSTUMER_ID)
);
CREATE SYNONYM DETAILS_STDKEVI3 FOR DETAIL_STDKEVI3;
